## §4 · Playwright Capture Script

Generate `<project_root>/scripts/capture_manual.py` with this structure:

```python
#!/usr/bin/env python3
"""
capture_manual.py — Auto-generated by doc_builder skill
Run: python scripts/capture_manual.py --base-url http://localhost:3000 --out public/doc/manual/assets
"""
import argparse
import asyncio
from pathlib import Path
from playwright.async_api import async_playwright

BASE_URL = "http://localhost:3000"
OUT_DIR = Path("public/doc/manual/assets")

# ── Auth helper ────────────────────────────────────────────────────────────────
async def login(page, role: str):
    """Login as a specific role. Adjust selectors for the target project."""
    credentials = {
        "admin":  {"email": "admin@test.com",  "password": "test1234"},
        "staff":  {"email": "staff@test.com",  "password": "test1234"},
        "public": None,
    }
    creds = credentials.get(role)
    if not creds:
        return
    await page.goto(f"{BASE_URL}/login")
    await page.fill('[name="email"]', creds["email"])
    await page.fill('[name="password"]', creds["password"])
    await page.click('[type="submit"]')
    await page.wait_for_load_state("networkidle")

# ── Capture targets ────────────────────────────────────────────────────────────
CAPTURES = [
    # { "file": "<filename>.png", "path": "/url", "role": "admin|staff|public",
    #   "action": None | "click:#selector" | "fill:#sel=value" | "wait:.selector" }
    {"file": "index_overview.png",         "path": "/",             "role": "public", "action": None},
    # ── FILL REMAINING FROM §3 SCREENSHOT PLAN ──
    # {"file": "admin_dashboard_filled.png", "path": "/admin",        "role": "admin",  "action": "wait:.data-table"},
    # {"file": "admin_users_add_modal.png",  "path": "/admin/users",  "role": "admin",  "action": "click:#add-user-btn"},
]

# ── Runner ────────────────────────────────────────────────────────────────────
async def capture_all(base_url: str, out_dir: Path):
    out_dir.mkdir(parents=True, exist_ok=True)
    async with async_playwright() as pw:
        browser = await pw.chromium.launch()
        current_role = None
        context = None
        page = None

        for cap in CAPTURES:
            role = cap["role"]
            if role != current_role:
                if context:
                    await context.close()
                context = await browser.new_context(viewport={"width": 1280, "height": 800})
                page = await context.new_page()
                await login(page, role)
                current_role = role

            await page.goto(f"{base_url}{cap['path']}")
            await page.wait_for_load_state("networkidle")

            action = cap.get("action")
            if action:
                op, selector = action.split(":", 1)
                if op == "click":
                    await page.click(selector)
                    await page.wait_for_timeout(500)
                elif op == "wait":
                    await page.wait_for_selector(selector)
                elif op == "fill":
                    sel, val = selector.split("=", 1)
                    await page.fill(sel.strip(), val.strip())

            out_path = out_dir / cap["file"]
            await page.screenshot(path=str(out_path), full_page=False)
            print(f"  ✓ {cap['file']}")

        if context:
            await context.close()
        await browser.close()

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--base-url", default=BASE_URL)
    parser.add_argument("--out", default=str(OUT_DIR))
    args = parser.parse_args()
    asyncio.run(capture_all(args.base_url, Path(args.out)))
```
